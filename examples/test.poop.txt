class Animal {
  String name;

  init(String name) {
    this.name = name;
  }

  method speak() Void {
    println(this.name);
    println("animal");
    return;
  }
}

class Cat extends Animal {
  init(String name) {
    super();
    this.name = name;
  }

  method speak() Void {
    println(this.name);
    println("cat");
    return;
  }
}

class Dog extends Animal {
  init(String name) {
    super();
    this.name = name;
  }

  method speak() Void {
    println(this.name);
    println("dog");
    return;
  }
}
