class Animal {

  init() {}

  method speak() Void {
    println("animal");
    return;
  }

}

class Cat extends Animal {

  init() {
    super();
  }

  method speak() Void {
    println("cat");
    return;
  }

}

class Dog extends Animal {

  init() {
    super();
  }

  method speak() Void {
    println("dog");
    return;
  }

}

Animal cat;
Animal dog;

cat = new Cat();
dog = new Dog();

cat.speak();
dog.speak();
